#+begin_src elisp :var links=links :results silent
(defun link (name)
  (let ((from (f-join (f-parent (buffer-file-name)) "linked" "home" name))
	(to   (f-join "~" name)))

    (unless (f-symlink? to)
      (if (f-file? to)
	  (error "File already exists: %s" to) 
	(f-symlink from to)))))

(-map (lambda (list) (link (car list))) links)
#+end_src
